class TextProcess(object):
        
    @staticmethod
    def lower_case(text):
        return text.lower()
    
    @staticmethod
    def strip_html_tags(text):
        soup = BeautifulSoup(text, "html.parser")
        stripped_text = soup.get_text()
        return stripped_text
    
    @staticmethod
    def remove_numbers(text):
        pattern = r'[^a-zA-z\s]'
        return re.sub(pattern, '', text)
        
    @staticmethod
    def tokenarize(text):
        return nltk.word_tokenize(text)
    
    @staticmethod
    def remove_stopwords(text, stop_words):
        new_text = []
        for word in text:
            if word not in stop_words:
                new_text.append(word)
        return new_text
    
    @staticmethod
    def normalize(text):
        text = TextProcess.lower_case(text)
        text = TextProcess.strip_html_tags(text)
        text = TextProcess.remove_numbers(text)
        text = TextProcess.tokenarize(text)
        text = TextProcess.remove_stopwords(text, stop_words)
        return text
    
    @staticmethod
    def lemmatize_verbs(text):
        lemmatizer = WordNetLemmatizer()
        lemmas = []
        for word in text:
            lemma = lemmatizer.lemmatize(word, pos='v')
            lemmas.append(lemma)
        return lemmas
    
    @staticmethod
    def stem_words(text):
        stemmer = LancasterStemmer()
        stems = []
        for word in text:
            stem = stemmer.stem(word)
            stems.append(stem)
        return stems
    
    @staticmethod
    def join_words(text):
        return ' '.join(text)

For normalization i did remove html tags if any, any number, any special character and the standard english stop words. I found that lemmas has a better understand of the short words or it didn't repleace any word with an strange one. Inside the text we also can found characters like @ and the names of the airlines but with this normalization process i was able to remove them. For example:
